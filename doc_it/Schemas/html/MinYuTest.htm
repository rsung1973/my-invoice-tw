CREATE VIEW dbo.MinYuTest
AS
SELECT         dbo.InvoiceAllowance.AllowanceID, dbo.InvoiceAllowance.SellerId, 
                          dbo.InvoiceItem.No, dbo.InvoiceAllowance.TotalAmount AS 折讓總金額, 
                          dbo.InvoiceAllowance.TaxAmount AS 折讓稅總金額, 
                          dbo.InvoiceAllowanceItem.Amount AS 折讓金額, 
                          dbo.InvoiceAllowanceItem.Tax AS 折讓稅率, 
                          dbo.InvoiceAmountType.TaxAmount AS 發票稅金額, 
                          dbo.InvoiceAmountType.SalesAmount AS 發票金額, 
                          dbo.InvoiceAmountType.TaxRate AS 發票稅率, 
                          dbo.InvoiceAmountType.TotalAmount AS 發票總金額, 
                          dbo.InvoiceBuyer.ReceiptNo, dbo.InvoiceBuyer.Name, 
                          dbo.InvoiceBuyer.Address, dbo.Organization.CompanyName
FROM             dbo.InvoiceItem INNER JOIN
                          dbo.InvoiceAllowanceItem INNER JOIN
                          dbo.InvoiceProductItem ON 
                          dbo.InvoiceAllowanceItem.ProductItemID = dbo.InvoiceProductItem.ItemID INNER
                           JOIN
                          dbo.InvoiceAllowanceDetails ON 
                          dbo.InvoiceAllowanceItem.ItemID = dbo.InvoiceAllowanceDetails.ItemID INNER JOIN
                          dbo.InvoiceAllowance ON 
                          dbo.InvoiceAllowanceDetails.AllowanceID = dbo.InvoiceAllowance.AllowanceID ON
                           dbo.InvoiceItem.InvoiceID = dbo.InvoiceAllowance.InvoiceID INNER JOIN
                          dbo.InvoiceAmountType ON 
                          dbo.InvoiceItem.InvoiceID = dbo.InvoiceAmountType.InvoiceID INNER JOIN
                          dbo.InvoiceBuyer ON 
                          dbo.InvoiceItem.InvoiceID = dbo.InvoiceBuyer.InvoiceID INNER JOIN
                          dbo.Organization ON 
                          dbo.InvoiceAllowance.SellerId = dbo.Organization.ReceiptNo
WHERE         (dbo.InvoiceItem.No = N'40477389') OR
                          (dbo.InvoiceItem.No = N'40477386')
