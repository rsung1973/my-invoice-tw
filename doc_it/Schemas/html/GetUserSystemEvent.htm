SET ANSI_NULLS ON
SET QUOTED_IDENTIFIER ON
Create PROCEDURE [InboxManager].[GetUserSystemEvent]
	(
		@OrgaCateID	int,
		@RoleID		int,
		@MessageID	int		=	18,
		@RecordCount	int	=	0 output
	)
AS
	/* SET NOCOUNT ON */
	SELECT UserInbox.MsgID, UserInbox.MsgDate, UserInbox.Sender, UserInbox.EventID, SystemEvent.EventDate, SystemEvent.Subject, 
				   SystemEvent.EventContent
	FROM  UserInbox INNER JOIN
				   SystemEvent ON UserInbox.EventID = SystemEvent.EventID
	WHERE (UserInbox.OrgaCateID = @OrgaCateID) AND (UserInbox.MessageID = @MessageID) AND (UserInbox.RoleID = @RoleID)
	ORDER BY SystemEvent.EventDate DESC

	set @RecordCount = @@ROWCOUNT
	
	RETURN
