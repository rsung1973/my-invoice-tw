CREATE VIEW dbo.viewBonusReportList
AS
SELECT dbo.InvoiceWinningNumber.NumberID, dbo.InvoiceWinningNumber.Year, dbo.InvoiceWinningNumber.MonthFrom, 
               dbo.InvoiceWinningNumber.MonthTo, dbo.InvoiceWinningNumber.WinningType, dbo.InvoiceItem.No, dbo.InvoiceItem.InvoiceDate, 
               dbo.Organization.CompanyName, dbo.Organization.CompanyID, dbo.Organization.ReceiptNo, dbo.Organization.Addr
FROM  dbo.InvoiceWinningNumber INNER JOIN
               dbo.InvoiceItem ON dbo.InvoiceWinningNumber.InvoiceID = dbo.InvoiceItem.InvoiceID INNER JOIN
               dbo.Organization ON dbo.InvoiceItem.SellerID = dbo.Organization.CompanyID AND dbo.InvoiceItem.DonationID = dbo.Organization.CompanyID
