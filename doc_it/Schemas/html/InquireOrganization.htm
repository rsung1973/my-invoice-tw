Create PROCEDURE [OrganizationManager].[InquireOrganization]
(
	@QueryCond	nvarchar(max)	=	null,
	@OrderBy	nvarchar(max)	=	Null,
	@RecordCount int = 0 output	
)
as


declare @SqlStmt	nvarchar(max)
declare @RecordCountStmt	nvarchar(max)
declare @SqlCmd	nvarchar(max)

declare @RecordCountCmd	nvarchar(max)

set @SqlCmd = N' SELECT *	FROM  Organization '
set @RecordCountCmd = N'select @RecordCount=count(*) from Organization '

if @OrderBy is Not Null
	set @OrderBy = N' Order by ' + @OrderBy
else
	set @OrderBy = N''

if @QueryCond is Not Null
	begin
		set @SqlStmt = 
			@SqlCmd + N' where ' + @QueryCond +  @OrderBy
		set @RecordCountStmt = 
			@RecordCountCmd + N' where ' + @QueryCond
	end
else
	begin
		set @SqlStmt = @SqlCmd + @OrderBy
		set @RecordCountStmt = @RecordCountCmd
	end

execute sp_executesql @SqlStmt		

execute sp_executesql @RecordCountStmt,
	N'	@RecordCount	int	output ',
		@RecordCount output
